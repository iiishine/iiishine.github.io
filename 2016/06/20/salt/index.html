发表于 2016-06-20


#centos7 安装saltstack#

服务端

安装epel-release和salt-master工具包
yum install epel-release -y
yum install salt-master -y
yum install python-jinja2 -y
配置
vim /etc/salt/master
    interface: 7.7.1.73
    file_roots:
    base:
        - /srv/salt
开机自启
systemctl enable salt-master.service
启动服务
systemctl start salt-master.service
查看端口
netstat -tunlp | grep python
ps -ef |grep salt
关闭防火墙
systemctl disable firewalld.service
systemctl stop  iptables.service
systemctl stop firewalld.service

客户端

安装epel-release工具包和salt-minion客户端
yum install epel-release -y
yum install salt-minion -y
yum install python-jinja2 -y
配置
vim /etc/salt/minion
    master: 7.7.1.73
    id: iiishine
开机自启
systemctl enable salt-minion.service
启动服务
systemctl start salt-minion.service
查看端口
ps -ef |grep salt
关闭防火墙
systemctl disable firewalld.service
systemctl stop  iptables.service
systemctl stop firewalld.service

证书认证

salt-key -L            列出所有密钥
salt-key -a iiishine    使用id event 接受密钥
salt-key -d iiishine    删除event编码

写简单sls文件

mkdir /src/salt/
[root@iiishine salt]# cat top.sls 
base:
  'iiishine':
    - iiishine
[root@iiishine salt]# cat iiishine.sls 
httpd:
  pkg:
    - installed

[root@iiishine salt]# salt 'iiishine' state.highstate
